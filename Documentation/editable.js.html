KIP.Objects.Editable = function (id, type, content, validate, update) {
  this.id = id;
  this.type = type || "text";
  this.content = content;
  this.validate = validate;
  this.update = update;
  this.is_modifying = false;

  KIP.Objects.Drawable.call(this, this.id, "editable", this.content);

  this.CreateElements();
  this.AddListeners();
};
